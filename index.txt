    <p>Esta é uma calculadora de latência de RAM. Insira a taxa de dados (MT/s) e a latência CAS (CL),
        em seguida, ela ira calcular a latência absoluta para acessos à memória DDR em nanossegundos.</p>

    <script>
        function calcMemoryLatency(dataRate, casLatency) {
            if (dataRate < 1.0) {
                dataRate = 1.0;
            }
            if (casLatency < 1.0) {
                casLatency = 1.0;
            }

            return (1.0 / (dataRate / 2.0) * casLatency) * 1000.0;
        }

        function doUpdate() {
            var dataRate = parseFloat(document.getElementById('ddr_mts').value);
            var casLatency = parseFloat(document.getElementById('ddr_cl').value);
            var result = calcMemoryLatency(dataRate, casLatency);
            document.getElementById('ddr_result').innerText = result.toFixed(2);
        }
    </script>

    <div style="margin-left: 1.5em;">
        <p>
            <b>MT/s:</b>
            <input style="width: 6em;" type="number" onchange="doUpdate();" id="ddr_mts" value="3000">
            <b style="margin-left: 1em;">CL:</b>
            <input style="width: 3.5em;" type="number" onchange="doUpdate();" id="ddr_cl" value="15">
            <br><br>
            Calculado em nanosegundos:
        </p>
        <div id="ddr_result" class="result-container">10</div>
